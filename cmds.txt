mkdir MVCapp
cd MVCapp
dotnet new classlib -n MVCapp.DAL --framework net8.0
dotnet new classlib -n MVCapp.BLL --framework net8.0
dotnet new mvc -n MVCapp.PL --framework net8.0

dotnet new sln -n MVCappSolution

dotnet sln add MVCapp.PL/MVCapp.PL.csproj
dotnet sln add MVCapp.BLL/MVCapp.BLL.csproj
dotnet sln add MVCapp.DAL/MVCapp.DAL.csproj

dotnet add MVCapp.PL/MVCapp.PL.csproj reference MVCapp.BLL/MVCapp.BLL.csproj
dotnet add MVCapp.BLL/MVCapp.BLL.csproj reference MVCapp.DAL/MVCapp.DAL.csproj

dotnet restore
dotnet build

dotnet run --project MVCapp.PL


cd MVCapp.DAL
dotnet add package Microsoft.EntityFrameworkCore --version 8.0.11 
dotnet add package Microsoft.EntityFrameworkCore.SqlServer  --version 8.0.11 
dotnet add package Microsoft.EntityFrameworkCore.Tools  --version 8.0.11 
dotnet add package Microsoft.EntityFrameworkCore.Design --version 8.0.11 
dotnet add package AutoMapper.Extensions.Microsoft.DependencyInjection

cd MVCapp.PL
dotnet add package Microsoft.EntityFrameworkCore.Design --version 8.0.11 


// ==================================================
// run this after setting up AppDbContext
cd <root-dir>
dotnet tool install --global dotnet-ef 


dotnet ef migrations add <mgration_name> --project MVCapp.DAL --startup-project MVCapp.PL
dotnet ef database update --project MVCapp.DAL --startup-project MVCapp.PL
